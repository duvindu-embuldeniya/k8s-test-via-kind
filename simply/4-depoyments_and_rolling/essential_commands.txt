//create a deployment...
kubectl create deployment nginx-deploy --image=nginx

//check it was ceated correctly...
kubectl get deployments

//describe it
kubectl describe deployment <deployment name>

//when create a deployment, replica also get created..check it...
kubectl get rs

//List the Pods created by that ReplicaSet...
kubectl get pods --selector=pod-template-hash=<value>




//To see the image version used by the deployments, run...
kubectl get deploy nginx-deploy -o jsonpath="{.spec.template.spec.containers[*].image}"

//update the image version...
kubectl set image deployment/nginx-deploy nginx=nginx:1.26.0

//chcek roll-out status...
kubectl rollout status deployment/nginx-deploy

//check roll-out history of a deployment...
kubectl rollout history deployment/nginx-deploy

//check in-detail about revision...
kubectl rollout history deployment/nginx-deploy --revision=1

//rollback...
kubectl rollout undo deployment/nginx-deploy --to-revision=1





//manual scale
kubectl scale deployment nginx-deploy --replicas=5

//auto scaling (horizontal pods autoscaler)
kubectl autoscale deployment nginx-deploy --cpu-percent=50 --min=2 --max=8
kubectl delete hpa nginx-deploy
