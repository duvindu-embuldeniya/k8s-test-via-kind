//create configmap + key:value
kubectl create configmap app-config --from-literal=APP_ENV=dev

kubectl get configmap

kubectl describe configmap app-configmap

//edit configmap values without touching image file
kubectl edit configmap app-configmap




//create secret
kubectl create secret generic db-secret --from-literal=DB_PASSWORD=admin123

kubectl get secret

kubectl describe secret db-secret




//apply yml file
kubectl apply -f pod.yml

//verify pod
kubectl get pods

//verify config map and secret correctly injected to pod
1|go inside pod
> kubectl exec -it config-secret-pod -- env

> kubectl exec -it config-secret-pod -- printenv APP_ENV
> kubectl exec -it config-secret-pod -- printenv DB_PASSWORD
